Build "Edubee Camp" — a comprehensive multi-operator educational camp
marketplace platform. Set up the complete project foundation, database,
middleware, auth system, and seed data.

═══════════════════════════════════════════
 PART 1: TECH STACK & PROJECT STRUCTURE
═══════════════════════════════════════════

Tech Stack:
  Frontend:  React 18 + TypeScript + Vite
  Styling:   Tailwind CSS + shadcn/ui
  Routing:   React Router v6
  Data:      TanStack Query v5 + axios
  i18n:      react-i18next
  Backend:   Node.js + Express + TypeScript
  ORM:       Drizzle ORM
  Database:  PostgreSQL
  Storage:   Supabase Storage (private buckets)
  Auth:      JWT (access 8h + refresh 7d)
  Email:     Nodemailer (SMTP)
  PDF:       @react-pdf/renderer
  ZIP:       archiver

Folder structure:

━━━ A. USERS & ACCESS ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

export const users = pgTable('users', {
  id:               uuid('id').primaryKey().defaultRandom(),
  email:            varchar('email', { length: 255 }).notNull().unique(),
  passwordHash:     varchar('password_hash', { length: 255 }).notNull(),
  role:             varchar('role', { length: 50 }).notNull(),
  -- ROLES:
  -- 'super_admin'    → Full platform control
  -- 'admin'          → Platform management, all data
  -- 'camp_coordinator' → EXTERNAL PARTNER: registers & operates own camps
  -- 'education_agent'  → Sales partner: consults clients, submits applications
  -- 'partner_institute'→ Service partner: provides educational programs
  -- 'partner_hotel'    → Service partner: provides accommodation
  -- 'partner_pickup'   → Service partner: provides transportation
  -- 'partner_tour'     → Service partner: provides tour activities
  -- 'parent_client'    → End customer: monitors child's program
  fullName:         varchar('full_name', { length: 255 }).notNull(),
  phone:            varchar('phone', { length: 50 }),
  whatsapp:         varchar('whatsapp', { length: 50 }),
  lineId:           varchar('line_id', { length: 100 }),
  avatarUrl:        varchar('avatar_url', { length: 500 }),
  timezone:         varchar('timezone', { length: 100 }).default('Asia/Seoul'),
  preferredLang:    varchar('preferred_lang', { length: 10 }).default('en'),
  -- camp_coordinator specific fields:
  companyName:      varchar('company_name', { length: 255 }),
  businessRegNo:    varchar('business_reg_no', { length: 100 }),
  countryOfOps:     varchar('country_of_ops', { length: 10 }),
  platformCommRate: decimal('platform_comm_rate', { precision: 5, scale: 2 }),
  -- % platform takes from coordinator's sales (default 10%)
  status:           varchar('status', { length: 20 }).default('pending'),
  -- 'active'|'inactive'|'pending'
  lastLoginAt:      timestamp('last_login_at'),
  createdAt:        timestamp('created_at').defaultNow(),
  updatedAt:        timestamp('updated_at').defaultNow(),
});

export const pagePermissions = pgTable('page_permissions', {
  id:        uuid('id').primaryKey().defaultRandom(),
  role:      varchar('role', { length: 50 }).notNull(),
  pageSlug:  varchar('page_slug', { length: 100 }).notNull(),
  canAccess: boolean('can_access').default(false),
  createdAt: timestamp('created_at').defaultNow(),
}, (t) => ({ uniq: unique().on(t.role, t.pageSlug) }));

export const fieldPermissions = pgTable('field_permissions', {
  id:        uuid('id').primaryKey().defaultRandom(),
  role:      varchar('role', { length: 50 }).notNull(),
  tableName: varchar('table_name', { length: 100 }).notNull(),
  fieldName: varchar('field_name', { length: 100 }).notNull(),
  canView:   boolean('can_view').default(false),
  canEdit:   boolean('can_edit').default(false),
  createdAt: timestamp('created_at').defaultNow(),
}, (t) => ({ uniq: unique().on(t.role, t.tableName, t.fieldName) }));

export const impersonationLogs = pgTable('impersonation_logs', {
  id:           uuid('id').primaryKey().defaultRandom(),
  actorUserId:  uuid('actor_user_id').references(() => users.id),
  targetUserId: uuid('target_user_id').references(() => users.id),
  startedAt:    timestamp('started_at').defaultNow(),
  endedAt:      timestamp('ended_at'),
  pagesVisited: jsonb('pages_visited'),
  ipAddress:    varchar('ip_address', { length: 50 }),
});

/server/src/middleware/partnerScope.ts
  ★ CRITICAL: Enforces data isolation for ALL partner roles
  Runs after authenticate for every protected route.

  Sets req.scopeFilter based on role:

  camp_coordinator:
    packageGroups → WHERE camp_provider_id = userId
    contracts     → WHERE camp_provider_id = userId
    applications  → WHERE packageGroup.camp_provider_id = userId
    *_mgt         → via contract WHERE camp_provider_id = userId
    settlementMgt → WHERE provider_user_id = userId
                    OR contract.camp_provider_id = userId
    invoices      → WHERE contract.camp_provider_id = userId

  education_agent:
    leads         → WHERE agent_id = userId
    applications  → WHERE agent_id = userId
    contracts     → via application WHERE agent_id = userId

  partner_institute:
    institute_mgt → WHERE institute_id = userId
    contracts     → via institute_mgt WHERE institute_id = userId
    -- student name/grade/medical visible, NO financial/passport

  partner_hotel:
    hotel_mgt     → WHERE hotel_id = userId
    -- guest name/dates/dietary visible, NO financial/passport

  partner_pickup:
    pickup_mgt    → WHERE driver_id = userId
    -- student name/location/time visible, NO personal/financial

  partner_tour:
    tour_mgt      → WHERE tour_company_id = userId
    -- participant count/schedule visible, NO personal/financial

  parent_client:
    applications  → WHERE client_id = userId
    contracts     → via application WHERE client_id = userId

  SECURITY: Backend ALWAYS validates scope.
  Client-side filters are UI convenience only.
  NEVER return out-of-scope data regardless of query params.
