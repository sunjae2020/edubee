// GET /api/package-groups/:id/products

// POST /api/package-groups/:id/products
router.post('/:id/products',
  authenticate,
  requireRole('super_admin', 'admin', 'camp_coordinator'),
  async (req, res) => {
    const { productId, quantity = 1, unitPrice } = req.body;
    if (!productId)
      return res.status(400).json({ success: false, error: 'productId required' });
    try {
      // camp_coordinator: 자신의 package group에만 연결 가능
      if (req.user.role === 'camp_coordinator') {
        const [pg] = await db.select().from(packageGroups).where(
          and(eq(packageGroups.id, req.params.id),
              eq(packageGroups.campProviderId, req.user.id))
        );
        if (!pg) return res.status(403).json({ success: false, error: 'Access denied' });
      }
      // 중복 연결 방지
      const [existing] = await db.select().from(packageGroupProducts).where(
        and(eq(packageGroupProducts.packageGroupId, req.params.id),
            eq(packageGroupProducts.productId, productId))
      );
      if (existing)
        return res.status(409).json({ success: false, error: 'Already linked' });

      const [prod] = await db.select().from(products).where(eq(products.id, productId));
      if (!prod) return res.status(404).json({ success: false, error: 'Product not found' });

      const [link] = await db.insert(packageGroupProducts).values({
        packageGroupId: req.params.id,
        productId,
        quantity,
        unitPrice: unitPrice ?? prod.cost,
      }).returning();
      res.json({ success: true, data: link });
    } catch {
      res.status(500).json({ success: false, error: 'Failed to link product' });
    }
  }
);

    }
  }
);

// GET /api/users/:id/products
// 특정 Service Provider(Partner)가 제공하는 상품 목록 반환
router.get('/:id/products', authenticate, async (req, res) => {
  // 접근 제어: SA/AD 또는 본인만
  const isSelf   = req.user.id === req.params.id;
  const isAdmin  = ['super_admin', 'admin'].includes(req.user.role);
  if (!isSelf && !isAdmin)
    return res.status(403).json({ success: false, error: 'Access denied' });

  try {
    const prods = await db.select().from(products)
      .where(and(
        eq(products.providerAccountId, req.params.id),
        eq(products.status, 'active')
      ))
      .orderBy(products.productType, products.productName);

    // 각 product의 연결된 package groups를 함께 조회
    const result = await Promise.all(prods.map(async (p) => {
      const linkedGroups = await db
        .select({ id: packageGroups.id, nameEn: packageGroups.nameEn,
                  countryCode: packageGroups.countryCode })
        .from(packageGroupProducts)
        .innerJoin(packageGroups,
          eq(packageGroupProducts.packageGroupId, packageGroups.id))
        .where(eq(packageGroupProducts.productId, p.id));
      return { ...p, linkedGroups };
    }));

    res.json({ success: true, data: result });
  } catch {
    res.status(500).json({ success: false, error: 'Failed to fetch user products' });
  }
});
