Create /server/src/services/reportAutoPopulate.ts

export async function autoPopulateSections(contractId: string): Promise<SectionData[]>

This function pulls data from existing service records and returns
structured content for each section type.

SECTION 1 — student_profile:
  Query:
    SELECT ap.*, a.package_id, a.preferred_start_date,
           p.name as package_name, pg.name_en as program_name,
           pg.location, c.start_date, c.end_date,
           u.full_name as coordinator_name, u.phone as coordinator_contact
    FROM applications a
    JOIN contracts c ON c.application_id = a.id
    JOIN application_participants ap ON ap.application_id = a.id
    JOIN packages p ON p.id = a.package_id
    JOIN package_groups pg ON pg.id = a.package_group_id
    JOIN users u ON u.id = c.camp_provider_id
    WHERE c.id = contractId
    AND ap.participant_type = 'primary_student'
    LIMIT 1

  Content shape:
  {
    fullName, fullNameNative, dateOfBirth,
    age: calculateAge(dateOfBirth),
    gender, nationality,
    passportNumber, passportExpiry,
    grade, schoolName, englishLevel,
    medicalConditions, dietaryRequirements, specialNeeds,
    emergencyPhone: ap.phone, whatsapp: ap.whatsapp,
    lineId: ap.line_id,
    programName, packageName,
    startDate: c.start_date, endDate: c.end_date,
    totalDays: daysBetween(c.start_date, c.end_date),
    campLocation: pg.location,
    coordinatorName, coordinatorContact
  }

SECTION 2 — pickup_schedule:
  Query: SELECT * FROM pickup_mgt WHERE contract_id = contractId
  Content shape:
  {
    pickups: [{
      pickupType, fromLocation, toLocation,
      pickupDatetime, vehicleInfo, driverNotes, status
    }]
  }

SECTION 3 — accommodation:
  Query: SELECT hm.*, u.full_name as hotel_name
         FROM hotel_mgt hm
         JOIN users u ON u.id = hm.hotel_id
         WHERE hm.contract_id = contractId
  Content shape:
  {
    hotelName, roomType, checkinDate, checkinTime,
    checkoutDate, checkoutTime, confirmationNo,
    guestNotes, status
  }

SECTION 4 — academic:
  Query: SELECT im.*, u.full_name as institute_name
         FROM institute_mgt im
         JOIN users u ON u.id = im.institute_id
         WHERE im.contract_id = contractId
  Content shape:
  {
    instituteName, programDetails,
    startDate, endDate, totalHours,
    englishLevelStart, englishLevelEnd,
    schedule: im.schedule ?? [],
    teacherComments, progressNotes, status
  }

SECTION 5 — tour:
  Query: SELECT tm.*, u.full_name as tour_company_name
         FROM tour_mgt tm
         JOIN users u ON u.id = tm.tour_company_id
         WHERE tm.contract_id = contractId
  Content shape:
  {
    tours: [{
      tourName, tourDate, startTime, endTime,
      meetingPoint, highlights, guideInfo, tourNotes, status
    }]
  }

SECTION 6 — summary:
  Content shape (empty template):
  {
    overallNotes: '',
    achievements: [],
    recommendations: '',
    closingMessage: 'Thank you for choosing Edubee Camp. We look forward to supporting your educational journey.'
  }

DEFAULT SECTION TITLES:
  student_profile  → "Student Profile"
  pickup_schedule  → "Pickup & Transportation"
  accommodation    → "Accommodation"
  academic         → "Academic Program"
  tour             → "Tours & Activities"
  summary          → "Program Summary"

DEFAULT DISPLAY ORDER: 0,1,2,3,4,5

 PART 3: REPORT ROUTES
═══════════════════════════════════════════

Create /server/src/routes/reports.ts
Mount at: app.use('/api/reports', authenticate, reportsRouter)

── GET /api/reports ─────────────────────────────────────────
Query params: contractId?, status?, page=1, limit=20

Scope filter (apply BEFORE query):
  super_admin / admin  → no filter
  camp_coordinator     → contracts WHERE camp_provider_id = req.user.id
  education_agent      → applications WHERE agent_id = req.user.id
                         AND status = 'published'
  parent_client        → applications WHERE client_id = req.user.id
                         AND status = 'published'
  others               → 403

Response: {
  data: [{
    id, contractId, contractNumber, reportTitle, status,
    generatedByName, publishedAt, summaryNotes,
    studentName, programName, createdAt, updatedAt,
    sectionsCount
  }],
  total, page, limit
}

── GET /api/reports/:id ─────────────────────────────────────
Same scope filter.
If status='draft' AND role is education_agent or parent_client → 403

Response: {
  ...reportFields,
  sections: [ ...orderedByDisplayOrder ]
}

── POST /api/reports ────────────────────────────────────────
Allowed: super_admin, admin, camp_coordinator (own contracts)
Body: { contractId: string, reportTitle?: string }

Validate:
  - Contract exists
  - CC: contract.camp_provider_id must equal req.user.id
  - No existing report for this contractId (unique per contract)

Actions:
  1. Auto-generate title if not provided:
     "[studentName] — [programName] Report"
  2. Create programReports record
  3. Call autoPopulateSections(contractId)
  4. Insert all 6 sections in order
  5. Return full report with sections

── PUT /api/reports/:id ─────────────────────────────────────
Allowed: super_admin, admin, camp_coordinator (own)
Body: { reportTitle?, summaryNotes? }

If status = 'published' → return 400:
  { error: 'Cannot edit a published report. Unpublish first.' }

── PATCH /api/reports/:id/publish ───────────────────────────
Allowed: super_admin, admin, camp_coordinator (own)
Actions:
  1. Set status='published', publishedAt=NOW()
  2. Create notification for agent (application.agent_id):
     {
       type: 'report_published',
       title: 'Program Report Available',
       message: `Program report for ${studentName} is now ready.`,
       referenceType: 'report',
       referenceId: reportId
     }
  3. Create notification for parent (application.client_id):
     same message
Return: { success: true, publishedAt }

── PATCH /api/reports/:id/unpublish ─────────────────────────
Allowed: super_admin, admin only
Set status='draft', publishedAt=NULL

── DELETE /api/reports/:id ──────────────────────────────────
Allowed: super_admin only
Soft delete: set deletedAt=NOW()

── PATCH /api/reports/:id/sections/:sectionId ───────────────
Allowed: super_admin, admin, camp_coordinator (own)
Body: { sectionTitle?, content?, isVisible?, displayOrder? }
content is freeform JSONB — accept any shape
Return: updated section

── POST /api/reports/:id/sections ───────────────────────────
Allowed: super_admin, admin, camp_coordinator (own)
Body: { sectionType: 'custom', sectionTitle: string, content?: object }
Validates: sectionType must be 'custom'
Sets displayOrder to max(existing) + 1
Return: new section

── DELETE /api/reports/:id/sections/:sectionId ──────────────
Allowed: super_admin, admin, camp_coordinator (own)
Validate: sectionType must be 'custom'
If not custom → 400: { error: 'Default sections cannot be deleted.' }

── PATCH /api/reports/:id/sections/reorder ──────────────────
Body: { orderedIds: string[] }
Updates displayOrder for all provided section IDs atomically
(use a transaction)

── POST /api/reports/:id/sync ───────────────────────────────
Allowed: super_admin, admin, camp_coordinator (own)
Re-runs autoPopulateSections(contractId)
For each section type:
  - If section.updatedAt equals createdAt (never manually edited)
    → overwrite content
  - Else → skip (preserve manual edits)
Return: {
  synced: string[],   // section types that were updated
  skipped: string[]   // section types with manual edits
}

── GET /api/reports/:id/pdf ─────────────────────────────────
Same scope as GET /:id (draft → 403 for EA/parent)
Generates PDF server-side using generateReportPdf(report, sections, role)
Returns binary PDF stream with headers:
  Content-Type: application/pdf
  Content-Disposition: attachment; filename="EdubeeCamp_Report_[name]_[year].pdf"
