── R2. PORT & HOST BINDING ──────────────────────────────────
Replit REQUIRES the server to bind to 0.0.0.0, NOT localhost or 127.0.0.1.
If bound to localhost, the Replit proxy cannot reach the server.

In server/src/index.ts use EXACTLY:

  app.listen(PORT, '0.0.0.0', () => {
    console.log(`Server running on port ${PORT}`);
  });

Never use: app.listen(PORT) without the host argument.
Never use: app.listen(PORT, 'localhost', ...)

── R3. DATABASE CONNECTION POOLING ─────────────────────────
Replit containers sleep after inactivity. When they wake, existing
PostgreSQL connections may be stale or closed.

Use pg-pool with reconnect-safe settings in db/client.ts:

  import { Pool } from 'pg';
  import { drizzle } from 'drizzle-orm/node-postgres';

  const pool = new Pool({
    connectionString: process.env.DATABASE_URL,
    max: 5,                  -- keep pool small (Replit memory limit)
    idleTimeoutMillis: 10000,
    connectionTimeoutMillis: 5000,
    allowExitOnIdle: false,
  });

  -- Handle unexpected pool errors (prevents crash on stale connection)
  pool.on('error', (err) => {
    console.error('Unexpected DB pool error:', err.message);
  });

  export const db = drizzle(pool);

If using Neon (serverless Postgres via Replit DB):
  Use @neondatabase/serverless driver instead of pg:
  import { neon } from '@neondatabase/serverless';
  import { drizzle } from 'drizzle-orm/neon-http';
  const sql = neon(process.env.DATABASE_URL!);
  export const db = drizzle(sql);

import { db } from '../../../db/client';
import { accountLedgerEntries, invoices, transactions, receipts } from '../../../db/schema';
import { eq, and, sum, sql } from 'drizzle-orm';

type LedgerEntryParams = {
  accountId: string;
  sourceType: string;
  sourceId: string;
  contractId?: string;
  entryType: 'debit' | 'credit';
  amount: number;
  currency?: string;
  originalAmount?: number;
  originalCurrency?: string;
  audEquivalent?: number;
  exchangeRateToAud?: number;
  status?: string;
  description?: string;
  entryDate: string;
  createdBy?: string;
};

export async function createLedgerEntry(params: LedgerEntryParams) {
  try {
    const [entry] = await db.insert(accountLedgerEntries)
      .values({
        accountId:        params.accountId,
        sourceType:       params.sourceType,
        sourceId:         params.sourceId,
        contractId:       params.contractId,
        entryType:        params.entryType,
        amount:           params.amount.toString(),
        currency:         params.currency ?? 'AUD',
        originalAmount:   params.originalAmount?.toString(),
        originalCurrency: params.originalCurrency,
        audEquivalent:    params.audEquivalent?.toString(),
        exchangeRateToAud: params.exchangeRateToAud?.toString(),
        status:           params.status ?? 'pending',
        description:      params.description,
        entryDate:        params.entryDate,
        createdBy:        params.createdBy,
      })
      .returning();
    return entry;
  } catch (e: any) {
    console.error('createLedgerEntry failed:', e.message);
    throw new Error(`Ledger entry creation failed: ${e.message}`);
  }
}

export async function confirmLedgerEntry(entryId: string) {
  try {
    await db.update(accountLedgerEntries)
      .set({ status: 'confirmed', updatedAt: new Date() })
      .where(eq(accountLedgerEntries.id, entryId));
  } catch (e: any) {
    console.error('confirmLedgerEntry failed:', e.message);
    throw e;
  }
}

export async function confirmLedgerEntriesBySource(
  sourceType: string,
  sourceId: string
) {
  try {
    await db.update(accountLedgerEntries)
      .set({ status: 'confirmed', updatedAt: new Date() })
      .where(and(
        eq(accountLedgerEntries.sourceType, sourceType),
        eq(accountLedgerEntries.sourceId,   sourceId),
      ));
  } catch (e: any) {
    console.error('confirmLedgerEntriesBySource failed:', e.message);
    throw e;
  }
}

export async function reverseLedgerEntry(
  originalEntryId: string,
  reason: string,
  createdBy?: string
) {
  try {
    const [original] = await db.select()
      .from(accountLedgerEntries)
      .where(eq(accountLedgerEntries.id, originalEntryId))
      .limit(1);
    if (!original) throw new Error('Ledger entry not found');

    -- Create reversal entry
    const [reversal] = await db.insert(accountLedgerEntries)
      .values({
        accountId:        original.accountId,
        sourceType:       original.sourceType,
        sourceId:         original.sourceId,
        contractId:       original.contractId,
        entryType:        original.entryType === 'debit' ? 'credit' : 'debit',
        amount:           original.amount,
        currency:         original.currency,
        audEquivalent:    original.audEquivalent,
        exchangeRateToAud: original.exchangeRateToAud,
        status:           'confirmed',
        description:      `REVERSAL: ${original.description} — ${reason}`,
        entryDate:        new Date().toISOString().split('T')[0],
        createdBy,
      })
      .returning();

    -- Mark original as reversed
    await db.update(accountLedgerEntries)
      .set({ status: 'reversed', updatedAt: new Date() })
      .where(eq(accountLedgerEntries.id, originalEntryId));

    return { originalEntryId, reversalEntryId: reversal.id };
  } catch (e: any) {
    console.error('reverseLedgerEntry failed:', e.message);
    throw e;
  }
}

export async function getAccountBalance(userId: string) {
  try {
    const rows = await db.select({
      entryType:    accountLedgerEntries.entryType,
      currency:     accountLedgerEntries.currency,
      totalAud:     sql<string>`SUM(aud_equivalent)`,
    })
    .from(accountLedgerEntries)
    .where(and(
      eq(accountLedgerEntries.accountId, userId),
      sql`status IN ('pending', 'confirmed')`,
    ))
    .groupBy(
      accountLedgerEntries.entryType,
      accountLedgerEntries.currency,
    );

    const credits = rows.filter(r => r.entryType === 'credit')
                        .reduce((s, r) => s + parseFloat(r.totalAud ?? '0'), 0);
    const debits  = rows.filter(r => r.entryType === 'debit')
                        .reduce((s, r) => s + parseFloat(r.totalAud ?? '0'), 0);
    return {
      totalCredits: credits,
      totalDebits:  debits,
      netBalance:   credits - debits,
    };
  } catch (e: any) {
    console.error('getAccountBalance failed:', e.message);
    throw e;
  }
}
