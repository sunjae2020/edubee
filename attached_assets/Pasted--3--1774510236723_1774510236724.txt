SCENARIO 1 — TEST-FLOW-001 (Jason Kim)
풀 파이프라인: submitted → reviewing → quoted
             → contracted → 서비스 모듈 3개 활성화
=====================================================

--- 1-1. submitted → reviewing 전환 ---
UPDATE camp_applications
SET application_status = 'reviewing', updated_at = NOW()
WHERE application_ref = 'TEST-FLOW-001'
RETURNING id, application_ref, application_status;

-- ✅ 확인: application_status = 'reviewing'

--- 1-2. Convert to Quote API 호출 ---
-- 위에서 반환된 id 사용:
-- POST /api/camp-applications/{id}/convert-to-quote

-- 응답 확인:
-- {
--   success: true,
--   quoteId: "uuid",
--   quoteRefNumber: "QTE-XXXXXXXX",
--   campApplicationId: "uuid"
-- }

-- DB 확인:
SELECT
  ca.application_ref,
  ca.application_status,
  ca.quote_id        IS NOT NULL AS "quote_id 존재",
  ca.quoted_at       IS NOT NULL AS "quoted_at 존재",
  q.quote_ref_number             AS "Quote 번호",
  q.quote_status                 AS "Quote 상태",
  q.camp_application_id = ca.id  AS "역방향 연결 확인"
FROM camp_applications ca
LEFT JOIN quotes q ON q.id = ca.quote_id
WHERE ca.application_ref = 'TEST-FLOW-001';

-- ✅ 확인 항목:
-- application_status = 'quoted'
-- quote_id 존재 = true
-- quoted_at 존재 = true
-- Quote 상태 = 'Draft'
-- 역방향 연결 확인 = true

PART D — 전체 3개 시나리오 최종 요약 확인
=====================================================

SELECT
  ca.application_ref          AS "신청서",
  ca.applicant_name           AS "신청자",
  ca.application_status       AS "현재 상태",
  COALESCE(q.quote_ref_number,
    '(없음)')                 AS "Quote",
  COALESCE(c.contract_number,
    '(없음)')                 AS "Contract",
  COALESCE(
    (SELECT COUNT(*)::text
     FROM tour_mgt
     WHERE contract_id = c.id),
    '0')                      AS "Tour MGT",
  COALESCE(
    (SELECT COUNT(*)::text
     FROM pickup_mgt
     WHERE contract_id = c.id),
    '0')                      AS "Pickup MGT",
  COALESCE(
    (SELECT COUNT(*)::text
     FROM accommodation_mgt
     WHERE contract_id = c.id),
    '0')                      AS "Accommodation MGT"
FROM camp_applications ca
LEFT JOIN quotes q    ON q.id = ca.quote_id
LEFT JOIN contracts c ON c.application_id = ca.id
WHERE ca.application_ref IN (
  'TEST-FLOW-001', 'TEST-FLOW-002', 'TEST-FLOW-003'
)
ORDER BY ca.application_ref;
