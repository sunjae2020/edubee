/**
 * Knowledge Base Store
 * - Loads documents from Google Docs or local files
 * - Chunks text for better retrieval
 * - Simple cosine similarity search (no paid vector DB needed)
 * - Context cache management for Gemini cost savings
 */

import { genAI, docsClient, driveClient } from "../index.js";
import fs from "fs";
import path from "path";

// In-memory knowledge base
let knowledgeBase = {
  documents: [],      // { id, title, source, content, chunks, lastUpdated }
  embeddings: [],     // { chunkId, docId, embedding, text }
  lastCacheRefresh: null,
  cachedSystemPrompt: null,
};

// ─── Document Loading ───────────────────────────────────────────────────

/**
 * Load a Google Doc by Document ID
 * Doc ID is the part after /d/ in the URL:
 * https://docs.google.com/document/d/[DOC_ID]/edit
 */
export async function loadGoogleDoc(docId, title) {
  try {
    const res = await docsClient.documents.get({ documentId: docId });
    const doc = res.data;
    const content = extractTextFromDoc(doc);

    return {
      id: docId,
      title: title || doc.title,
      source: "google_docs",
      sourceUrl: `https://docs.google.com/document/d/${docId}`,
      content,
      lastUpdated: new Date().toISOString(),
    };
  } catch (err) {
    console.error(`Failed to load Google Doc ${docId}:`, err.message);
    throw err;
  }
}

/**
 * Extract plain text from Google Docs API response
 */
function extractTextFromDoc(doc) {
  let text = "";
  const body = doc.body?.content || [];

  for (const element of body) {
    if (element.paragraph) {
      for (const elem of element.paragraph.elements || []) {
        if (elem.textRun?.content) {
          text += elem.textRun.content;
        }
      }
    } else if (element.table) {
      for (const row of element.table.tableRows || []) {
        for (const cell of row.tableCells || []) {
          for (const content of cell.content || []) {
            if (content.paragraph) {
              for (const elem of content.paragraph.elements || []) {
                if (elem.textRun?.content) {
                  text += elem.textRun.content + " ";
                }
              }
            }
          }
        }
        text += "\n";
      }
    }
  }
  return text.trim();
}

/**
 * Load a local .txt or .md file
 */
export function loadLocalFile(filePath, title) {
  const content = fs.readFileSync(filePath, "utf-8");
  return {
    id: path.basename(filePath),
    title: title || path.basename(filePath),
    source: "local_file",
    sourceUrl: filePath,
    content,
    lastUpdated: new Date().toISOString(),
  };
}

// ─── Chunking ──────────────────────────────────────────────────────────

/**
 * Split document into overlapping chunks for better retrieval
 */
function chunkText(text, chunkSize = 800, overlap = 100) {
  const chunks = [];
  const sentences = text.split(/(?<=[.!?])\s+/);
  let current = "";
  let chunkIndex = 0;

  for (const sentence of sentences) {
    if ((current + sentence).length > chunkSize && current.length > 0) {
      chunks.push({ index: chunkIndex++, text: current.trim() });
      // Keep last overlap chars for context continuity
      const words = current.split(" ");
      current = words.slice(-Math.floor(overlap / 5)).join(" ") + " " + sentence;
    } else {
      current += " " + sentence;
    }
  }
  if (current.trim()) {
    chunks.push({ index: chunkIndex, text: current.trim() });
  }
  return chunks;
}

// ─── Embeddings ────────────────────────────────────────────────────────

/**
 * Generate embeddings using Gemini text-embedding-004 (FREE)
 */
async function generateEmbedding(text) {
  const embeddingModel = genAI.getGenerativeModel({
    model: "text-embedding-004",
  });
  const result = await embeddingModel.embedContent(text);
  return result.embedding.values;
}

/**
 * Cosine similarity between two vectors
 */
function cosineSimilarity(vecA, vecB) {
  let dot = 0, normA = 0, normB = 0;
  for (let i = 0; i < vecA.length; i++) {
    dot += vecA[i] * vecB[i];
    normA += vecA[i] ** 2;
    normB += vecB[i] ** 2;
  }
  return dot / (Math.sqrt(normA) * Math.sqrt(normB));
}

// ─── Knowledge Base Management ─────────────────────────────────────────

/**
 * Add a document to the knowledge base and generate embeddings
 */
export async function addDocument(docData) {
  const chunks = chunkText(docData.content);
  const docWithChunks = { ...docData, chunks, chunkCount: chunks.length };

  // Remove existing doc if updating
  knowledgeBase.documents = knowledgeBase.documents.filter(
    (d) => d.id !== docData.id
  );
  knowledgeBase.embeddings = knowledgeBase.embeddings.filter(
    (e) => e.docId !== docData.id
  );

  knowledgeBase.documents.push(docWithChunks);

  // Generate embeddings for all chunks
  console.log(`📊 Generating embeddings for "${docData.title}" (${chunks.length} chunks)...`);
  for (const chunk of chunks) {
    const embedding = await generateEmbedding(chunk.text);
    knowledgeBase.embeddings.push({
      chunkId: `${docData.id}-${chunk.index}`,
      docId: docData.id,
      text: chunk.text,
      embedding,
    });
  }

  // Rebuild cached system prompt
  rebuildSystemPrompt();
  console.log(`✅ Added "${docData.title}" to knowledge base`);
  return docWithChunks;
}

/**
 * Remove a document from knowledge base
 */
export function removeDocument(docId) {
  knowledgeBase.documents = knowledgeBase.documents.filter((d) => d.id !== docId);
  knowledgeBase.embeddings = knowledgeBase.embeddings.filter((e) => e.docId !== docId);
  rebuildSystemPrompt();
}

/**
 * Retrieve top-K most relevant chunks for a query
 */
export async function retrieveContext(query, topK = 4) {
  if (knowledgeBase.embeddings.length === 0) return [];

  const queryEmbedding = await generateEmbedding(query);

  const scored = knowledgeBase.embeddings.map((item) => ({
    ...item,
    score: cosineSimilarity(queryEmbedding, item.embedding),
  }));

  scored.sort((a, b) => b.score - a.score);
  return scored.slice(0, topK);
}

/**
 * Build the full system prompt from all documents
 * Used for context caching (all docs sent once, cached by Gemini)
 */
function rebuildSystemPrompt() {
  if (knowledgeBase.documents.length === 0) {
    knowledgeBase.cachedSystemPrompt = null;
    return;
  }

  const docContent = knowledgeBase.documents
    .map((doc) => `=== ${doc.title} ===\n${doc.content}`)
    .join("\n\n---\n\n");

  knowledgeBase.cachedSystemPrompt = `당신은 Edubee Camp 플랫폼의 전용 내부 도우미입니다.

## 절대적 규칙
1. 반드시 아래 [내부 문서]에 있는 정보만을 기반으로 답변하세요
2. 문서에 없는 내용은 절대 추측하거나 외부 지식으로 보완하지 마세요
3. 주식, 날씨, 뉴스, 플랫폼 외부 정보에 대한 질문에는 답변하지 마세요
4. 한국어로 질문하면 한국어로, 영어로 질문하면 영어로 답변하세요
5. 문서에서 관련 내용을 찾을 수 없으면 반드시 이렇게 말하세요:
   "죄송합니다. 해당 내용은 현재 등록된 문서에서 찾을 수 없습니다. 담당자(support@edubee.com)에게 문의해 주세요."

## 답변 스타일
- 명확하고 간결하게 답변하세요
- 관련 있는 경우 어느 문서의 정보인지 언급하세요
- 복잡한 내용은 번호 목록으로 정리하세요

[내부 문서]
${docContent}`;

  knowledgeBase.lastCacheRefresh = new Date().toISOString();
}

export function getKnowledgeBase() {
  return {
    documents: knowledgeBase.documents.map((d) => ({
      id: d.id,
      title: d.title,
      source: d.source,
      sourceUrl: d.sourceUrl,
      chunkCount: d.chunkCount,
      lastUpdated: d.lastUpdated,
    })),
    totalChunks: knowledgeBase.embeddings.length,
    hasSystemPrompt: !!knowledgeBase.cachedSystemPrompt,
    lastCacheRefresh: knowledgeBase.lastCacheRefresh,
  };
}

export function getSystemPrompt() {
  return knowledgeBase.cachedSystemPrompt;
}
