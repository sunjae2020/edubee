  ADD COLUMN allow_extra_upload boolean DEFAULT true;
-- When true: users with upload permission can add documents
-- outside the defined category list for that group.
-- When false: only the explicitly listed categories are allowed.

── 1.2 Add new table: document_extra_categories ─────────────

-- Stores ad-hoc categories created during extra uploads.
-- These are per-reference (not global), so they don't pollute
-- the master document_categories table.

export const documentExtraCategories = pgTable('document_extra_categories', {
  id:            uuid('id').primaryKey().defaultRandom(),
  referenceType: varchar('reference_type', { length: 20 }).notNull(),
  -- 'application'|'contract'|'institute_mgt'|'hotel_mgt'
  -- |'pickup_mgt'|'tour_mgt'|'user'
  referenceId:   uuid('reference_id').notNull(),
  categoryName:  varchar('category_name', { length: 255 }).notNull(),
  -- Free-text name entered by uploader
  categoryGroup: varchar('category_group', { length: 50 }).default('other'),
  -- Always stored as 'other' so existing permission rules apply
  createdBy:     uuid('created_by').references(() => users.id),
  createdAt:     timestamp('created_at').defaultNow(),
});

-- Add FK column to existing documents table (migration, not rebuild):
ALTER TABLE documents
  ADD COLUMN extra_category_id uuid
    REFERENCES document_extra_categories(id);
-- NULL = uses standard categoryId; NOT NULL = uses extra category.
-- A document uses exactly ONE of categoryId or extra_category_id.

── 1.3 Add service-level document linking ───────────────────

-- Services (institute_mgt, hotel_mgt, pickup_mgt, tour_mgt)
-- already link to contracts. Documents already link to contracts.
-- We need a direct view-level association for the service
-- Documents tab to filter relevant documents efficiently.

-- Add optional service_reference columns to documents table:
ALTER TABLE documents
  ADD COLUMN service_type varchar(20),
  -- 'institute'|'hotel'|'pickup'|'tour' (null if not service-linked)
  ADD COLUMN service_id uuid;
  -- FK to respective *_mgt table (polymorphic, no hard FK)

── 1.4 Add allow_extra_upload to doc_permissions tables ─────

ALTER TABLE default_doc_permissions
  ADD COLUMN can_upload_extra boolean DEFAULT false;

ALTER TABLE document_permissions
  ADD COLUMN can_upload_extra boolean DEFAULT false;

-- Default extra upload rights (seed this):
-- SA/AD/CC: can_upload_extra = true for ALL category groups
-- EA: can_upload_extra = true for personal, school, travel
-- Partners: can_upload_extra = true for their specific group only
--   partner_institute → school group only
--   partner_hotel     → travel group only
--   partner_pickup    → travel group only
--   partner_tour      → travel group only
-- parent_client: can_upload_extra = false (no extra uploads)

── 2.3 NEW endpoint: GET /api/documents/by-entity ───────────

Purpose: Powers the new Documents tab on any entity page.
Auth: Required (Bearer token). Enforces partnerScope.

Query params:
  entityType: 'application'|'contract'|'institute_mgt'
              |'hotel_mgt'|'pickup_mgt'|'tour_mgt'|'user'
  entityId: uuid

Logic:
  1. Resolve which referenceType/referenceId to query:
     - For application/contract: direct match
     - For service types (institute_mgt etc.):
       • Load the service record → get contract_id
       • Return documents WHERE referenceType='contract'
         AND referenceId=contract_id
         AND (service_type=entityType OR service_type IS NULL)
     - For user:
       • Return documents from ALL applications and contracts
         associated with this user (as client, agent, or provider)
       • Filter by can_view permission for the requesting role

  2. Apply role-based visibility:
     • Filter each document by documentPermissions or
       defaultDocPermissions (same logic as existing list endpoint)
     • Never return documents the role cannot view

  3. Return grouped by category_group:
     [
       {
         group: 'personal',
         groupLabel: 'Personal Documents',
         documents: [ ...doc objects with permissions ]
       },
       ...
       {
         group: 'other',
         groupLabel: 'Additional Documents',
         documents: [ ...extra uploaded docs ]
       }
     ]

── 2.4 NEW endpoint: GET /api/documents/permissions-check ───

Purpose: Frontend uses this before showing the upload button.

Query params:
  entityType: string
  entityId: uuid
  categoryGroup: string

Returns:
  {
    canUpload: boolean,
    canUploadExtra: boolean,
    availableCategories: [ { id, categoryCode, categoryNameEn, ... } ]
    -- Only categories visible to this role for this group
  }
